PYTHON — model_router.py 
# Triple-model intelligent routing
def route_query(text: str) -> str:
    text_lower = text.lower()

    # phi3:mini — code & analysis
    if any(kw in text_lower for kw in
        ["code", "debug", "function", "error"]):
        return "phi3:mini"

    # gemma2 — deep knowledge
    if any(kw in text_lower for kw in
        ["explain", "why", "history", "science"]):
        return "gemma2"

    # llama3.2 — general fallback
    return "llama3.2"
PYTHON — tts_engine.py 
class EdgeTTSWorker(QThread):
    """Async TTS — GUI never blocks."""
    VOICE = "en-IN-NeerjaNeural"

    async def _speak(self, text: str):
        clean = _sanitize_for_tts(text)
        communicate = edge_tts.Communicate(
            clean, self.VOICE
        )
        await communicate.save(tmp_path)
        pygame.mixer.music.load(tmp_path)
        pygame.mixer.music.play()

SETUP GUIDE

Up & Running in Minutes

ISHA's self-install engine handles most of the work. Follow these steps for your first run.

1
INSTALL OLLAMA

Download and install Ollama — the local AI runtime that powers ISHA's on-device models.

→ ollama.ai (download for Windows / macOS / Linux)
2
PULL LOCAL MODELS

Pull all three models manually, or let ISHA's self-install engine do it automatically on first run.

ollama pull phi3:mini ollama pull gemma2 ollama pull llama3.2
3
DOWNLOAD ISHA

Clone the repository or download the single Python file directly to your machine.

git clone https://github.com/Akhack7/isha
4
RUN ISHA

ISHA auto-installs all pip dependencies on first launch. Just run and let it initialize.

python isha_v10.py
